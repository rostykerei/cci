3   import nl.rostykerei.cci.datastructure.DoublyLinkedList;
4   
5   import java.util.ArrayList;
6   import java.util.List;
7   
8   /**
9    * Doubly-linked list implementation.
10   *
11   * @param <T> the class of the objects in the list
12   * @author Rosty Kerei
13   */
14  public final class DoublyLinkedListImpl<T>
15          extends AbstractLinkedList<T> implements DoublyLinkedList<T> {
16  
17      /**
18       * Link to the previous node.
19       */
20      private DoublyLinkedList<T> previous;
21  
22      /**
23       * Link to the next node.
24       */
25      private DoublyLinkedList<T> next;
26  
27      /**
28       * Constructor.
29       *
30       * @param value node data value
31       */
32      public DoublyLinkedListImpl(final T value) {
33          this.setData(value);
34      }
35  
36      @Override
37      public DoublyLinkedList<T> setNext(final DoublyLinkedList<T> nextNode) {
38  
39          if (nextNode == null && this.next != null) {
40              DoublyLinkedList<T> n = this.next;
41              this.next = null;
42              n.setPrevious(null);
43  
44              return this;
45          }
46  
47          this.next = nextNode;
48  
49          if (nextNode != null && nextNode.getPrevious() != this) {
50              nextNode.setPrevious(this);
51          }
52  
53          return this;
54      }
55  
56      @Override
57      public DoublyLinkedList<T> getNext() {
58          return this.next;
59      }
60  
61      @Override
62      public DoublyLinkedList<T> setPrevious(
63              final DoublyLinkedList<T> previousNode) {
64  
65          if (previousNode == null && this.previous != null) {
66              DoublyLinkedList<T> p = this.previous;
67              this.previous = null;
68              p.setNext(null);
69  
70              return this;
71          }
72  
73          this.previous = previousNode;
74  
75          if (previousNode != null && previousNode.getNext() != this) {
76              previousNode.setNext(this);
77          }
78  
79          return this;
80      }
81  
82      @Override
83      public DoublyLinkedList<T> getPrevious() {
84          return this.previous;
85      }
86  
87      @Override
88      public boolean isFirst() {
89          return this.previous == null;
90      }
91  
92      @Override
93      public boolean isLast() {
94          return this.next == null;
95      }
96  
97      @Override
98      public List<T> toList() {
99          List<T> result = new ArrayList<>();
100         DoublyLinkedList<T> node = this;
101 
102         while (node != null) {
103             result.add(node.getData());
104             node = node.getNext();
105         }
106 
107         return result;
108     }
109 }
