5   /**

11   */

13  
14      /**
15       * First element.
16       */
17      private final F first;
18  
19      /**
20       * Second element.
21       */
22      private final S second;
23  
24      /**
25       * Pair constructor.
26       *
27       * @param firstElement  first element.
28       * @param secondElement second element.
29       */
30      public PairImpl(final F firstElement, final S secondElement) {
31          this.first = firstElement;
32          this.second = secondElement;
33      }
34

39  
40      @Override
