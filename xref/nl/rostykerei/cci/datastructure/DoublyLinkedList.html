3   /**
4    * Doubly-linked list.
5    *
6    * @param <T> the class of the objects in the list
7    * @author Rosty Kerei
8    */
9   public interface DoublyLinkedList<T> extends LinkedList<T> {
10  
11      /**
12       * Gets next node.
13       *
14       * @return next node or null if there is no next
15       */
16      DoublyLinkedList<T> getNext();
17  
18      /**
19       * Sets next node.
20       *
21       * @param nextNode next node
22       * @return self
23       */
24      DoublyLinkedList<T> setNext(DoublyLinkedList<T> nextNode);
25  
26      /**
27       * Gets previous node.
28       *
29       * @return previous node of null if there is no previous
30       */
31      DoublyLinkedList<T> getPrevious();
32  
33      /**
34       * Sets previous node.
35       *
36       * @param previousNode previous node
37       */
38      void setPrevious(DoublyLinkedList<T> previousNode);
39  
40      /**
41       * Checks if the node is the first one.
42       *
43       * @return true if the node is the first one, false otherwise
44       */
45      boolean isFirst();
46  
47      /**
48       * Checks if the node is the last one.
49       *
50       * @return true if the node is the last one, false otherwise
51       */
52      boolean isLast();
53  
54  }
