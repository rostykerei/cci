3   /**
4    * Interface for the basic queue data structure.
5    *
6    * @param <T> the class of the objects in the queue
7    * @author Rosty Kerei
8    */
9   public interface Queue<T> {
10  
11      /**
12       * Adds an item to the end of the queue.
13       *
14       * @param item item to add
15       */
16      void add(T item);
17  
18      /**
19       * Removes the first item from the queue.
20       *
21       * @throws java.util.NoSuchElementException if the queue is empty
22       * @return first item
23       */
24      T remove();
25  
26      /**
27       * Returns the first item of the queue.
28       *
29       * @return the first item
30       */
31      T peek();
32  
33      /**
34       * Checks if the queue is empty.
35       *
36       * @return true if the queue if empty, false otherwise
37       */
38      boolean isEmpty();
39  }
