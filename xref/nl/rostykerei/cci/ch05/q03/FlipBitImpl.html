3   /**
4    * Implementation of {@link FlipBit}.
5    *
6    * @author Rosty Kerei
7    */
8   public class FlipBitImpl implements FlipBit {
9   
10      /**
11       * {@inheritDoc}
12       */
13      @Override
14      public int flipBit(final int input) {
15          int result = 0;
16  
17          // Flips bit to 1, get max 1s count and save max(result)
18          for (int i = 0; i < Integer.SIZE; i++) {
19              result = Math.max(maxOnes(input | (1 << i)), result);
20          }
21  
22          return result;
23      }
24  
25      /**
26       * Ands x with x << 1 until it becomes 0.
27       * The number of needed iterations is actually the length
28       * of the longest consecutive sequence of 1s.
29       *
30       * @param input input integer
31       * @return the longest consecutive sequence of 1s
32       */
33      private int maxOnes(final int input) {
34          int count = 0;
35          int x = input;
36  
37          while (x != 0) {
38              x &= x << 1;
39              count++;
40          }
41  
42          return count;
43      }
44  }
