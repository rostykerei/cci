3   import java.util.BitSet;

5   /**
6    * A silly implementation of {@link Insertion} using {@link BitSet}.
7    *
8    * @author Rosty Kerei
9    */
10  public class InsertionBitSetImpl implements Insertion {
11  
12      /**
13       * {@inheritDoc}
14       */
15      @Override
16      public int insert(final int n, final int m, final byte i, final byte j) {
17          BitSet nBits = BitSet.valueOf(new long[]{n});
18          BitSet mBits = BitSet.valueOf(new long[]{m});
19  
20          for (byte k = i; k <= j; k++) {
21              nBits.set(k, mBits.get(k - i));
22          }
23  
24          return (int) nBits.toLongArray()[0];
25      }
26  }
