3   /**
4    * Implementation of {@link PairwiseSwap}. Shifts all even bits right,
5    * all odd bits left and then ORs both results.
6    *
7    * @author Rosty Kerei
8    */
9   public class PairwiseSwapImpl implements PairwiseSwap {
10  
11      /**
12       * Even bits mask (32-bits).
13       */
14      private static final int EVEN_MASK = 0b10101010101010101010101010101010;
15  
16      /**
17       * Odd bits mask (32-bits).
18       */
19      private static final int ODD_MASK = 0b01010101010101010101010101010101;
20  
21      /**
22       * {@inheritDoc}
23       */
24      @Override
25      public int swapBits(final int x) {
26          int evenBits = x & EVEN_MASK;
27          int oddBits = x & ODD_MASK;
28  
29          return evenBits >> 1 | oddBits << 1;
30      }
31  }
