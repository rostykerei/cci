3   /**
4    * Another implementation of {@link RotateMatrix}.
5    *
6    * @author Rosty Kerei
7    */
8   public final class RotateMatrixRoll implements RotateMatrix {
9   
10      @Override
11      public int[][] rotate(final int[][] image) {
12          final int n = image.length;
13  
14          for (int i = 0; i < n / 2; i++) {
15              for (int j = 0; j < (n + 1) / 2; j++) {
16                  int temp = image[i][j];
17  
18                  image[i][j] = image[n - 1 - j][i];
19                  image[n - 1 - j][i] = image[n - 1 - i][n - 1 - j];
20                  image[n - 1 - i][n - 1 - j] = image[j][n - 1 - i];
21                  image[j][n - 1 - i] = temp;
22              }
23          }
24  
25          return image;
26      }
27  }
