3   /**
4    * Implementation of {@link CheckPermutation} using an array.
5    *
6    * @author Rosty Kerei
7    */
8   public final class CheckPermutationArray implements CheckPermutation {
9   
10      /**
11       * 255 for ASCII, Character.MAX_VALUE for Unicode.
12       */
13      private static final int ARRAY_LENGTH = 255;
14  
15      @Override
16      public boolean checkPermutation(final String a, final String b) {
17          int l = a.length();
18  
19          if (l == b.length()) {
20  
21              int[] charMap = new int[ARRAY_LENGTH];
22  
23              for (int i = 0; i < l; i++) {
24                  charMap[a.charAt(i)]++;
25              }
26  
27              for (int i = 0; i < l; i++) {
28                  if (--charMap[b.charAt(i)] < 0) {
29                      return false;
30                  }
31              }
32  
33              return true;
34  
35          } else {
36              return false;
37          }
38      }
39  }
