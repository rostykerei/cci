3   import java.util.Arrays;
4   
5   /**
6    * Implementation of {@link CheckPermutation} using a sorted string.
7    *
8    * @author Rosty Kerei
9    */
10  public final class CheckPermutationSort implements CheckPermutation {
11  
12      @Override
13      public boolean checkPermutation(final String a, final String b) {
14          return a.length() == b.length() && sortString(a).equals(sortString(b));
15      }
16  
17      /**
18       * Sorts given string.
19       *
20       * @param s string to sort
21       * @return sorted string
22       */
23      private String sortString(final String s) {
24          char[] chars = s.toCharArray();
25          Arrays.sort(chars);
26          return new String(chars);
27      }
28  }
