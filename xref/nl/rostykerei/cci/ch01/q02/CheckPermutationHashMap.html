3   import java.util.HashMap;
4   import java.util.Map;
5   
6   /**
7    * Implementation of {@link CheckPermutation} using an hash-map.
8    *
9    * @author Rosty Kerei
10   */
11  public final class CheckPermutationHashMap implements CheckPermutation {
12  
13      @Override
14      public boolean checkPermutation(final String a, final String b) {
15          int l = a.length();
16  
17          if (l == b.length()) {
18              Map<Character, Integer> charMapA = new HashMap<>();
19              Map<Character, Integer> charMapB = new HashMap<>();
20  
21              for (int i = 0; i < l; i++) {
22                  charMapA.merge(a.charAt(i), 1, (x, y) -> x + 1);
23                  charMapB.merge(b.charAt(i), 1, (x, y) -> x + 1);
24              }
25  
26              return charMapA.equals(charMapB);
27          } else {
28              return false;
29          }
30      }
31  }
