3   /**
4    * Implementation of {@link StringCompression}.
5    *
6    * @author Rosty Kerei
7    */
8   public final class StringCompressionImpl implements StringCompression {
9   
10      @Override
11      public String compress(final String input) {
12          StringBuilder output = new StringBuilder();
13          char[] chars = input.toCharArray();
14  
15          int count = 1;
16  
17          for (int i = 0; i < chars.length; i++) {
18              if (i == chars.length - 1 || chars[i] != chars[i + 1]) {
19                  output.append(chars[i]);
20                  output.append(count);
21                  count = 1;
22              } else {
23                  count++;
24              }
25          }
26  
27          if (output.length() > input.length()) {
28              return input;
29          } else {
30              return output.toString();
31          }
32      }
33  }
