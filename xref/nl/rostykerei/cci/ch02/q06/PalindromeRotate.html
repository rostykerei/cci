3   import nl.rostykerei.cci.datastructure.DoublyLinkedList;
4   import nl.rostykerei.cci.datastructure.impl.DoublyLinkedListImpl;
5   
6   /**
7    * Implementation of {@link Palindrome} using a rotated copy of the list.
8    * Takes O(N) extra memory but can be used for singly linked lists.
9    *
10   * @param <T> the class of the objects in the list
11   * @author Rosty Kerei
12   */
13  public final class PalindromeRotate<T> implements Palindrome<T> {
14  
15      @Override
16      public boolean isPalindrome(final DoublyLinkedList<T> input) {
17  
18          DoublyLinkedList<T> head = input;
19          DoublyLinkedList<T> rotated = rotateList(head);
20  
21          while (head != null) {
22              if (!head.getData().equals(rotated.getData())) {
23                  return false;
24              }
25  
26              head = head.getNext();
27              rotated = rotated.getNext();
28          }
29  
30          return true;
31      }
32  
33      /**
34       * Rotates list.
35       *
36       * @param input head node
37       * @return rotated list
38       */
39      private DoublyLinkedList<T> rotateList(final DoublyLinkedList<T> input) {
40          DoublyLinkedList<T> head = input;
41  
42          DoublyLinkedList<T> rotated = null;
43          DoublyLinkedList<T> prev = null;
44  
45          while (head != null) {
46              rotated = new DoublyLinkedListImpl<>(head.getData());
47  
48              if (prev != null) {
49                  rotated.setNext(prev);
50              }
51  
52              prev = rotated;
53  
54              head = head.getNext();
55          }
56  
57          return rotated;
58      }
59  }
