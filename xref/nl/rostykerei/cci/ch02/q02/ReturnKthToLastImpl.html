5   /**
6    * Implementation of {@link ReturnKthToLast}.
7    *
8    * @param <T> the class of the objects in the list
9    * @author Rosty Kerei
10   */
11  public final class ReturnKthToLastImpl<T> implements ReturnKthToLast<T> {
12  
13      @Override
14      public SinglyLinkedList<T> kthToLast(final SinglyLinkedList<T> list,
15                                           final int k) {
16  
17          SinglyLinkedList<T> left = list;
18          SinglyLinkedList<T> right = list;
19  
20          int i = 0;
21          while (right != null) {
22              if (i > k) {
23                  left = left.getNext();
24              }
25  
26              right = right.getNext();
27              i++;
28          }
29  
30          if (i > k) {
31              return left;
32          }
33  
34          return null;
35      }
36  
37  }
