5   import java.util.HashSet;
6   import java.util.Set;
7   
8   /**
9    * Implementation of {@link LoopDetection} using a hash-set.
10   *
11   * @param <T> the class of the objects in the list
12   * @author Rosty Kerei
13   */
14  public final class LoopDetectionHashSet<T> implements LoopDetection<T> {
15  
16      @Override
17      public DoublyLinkedList<T> detectLoop(final DoublyLinkedList<T> list) {
18          Set<DoublyLinkedList<T>> set = new HashSet<>();
19  
20          DoublyLinkedList<T> head = list;
21  
22          while (head.getNext() != null) {
23              if (!set.add(head)) {
24                  return head;
25              }
26  
27              head = head.getNext();
28          }
29  
30          return null;
31      }
32  }
