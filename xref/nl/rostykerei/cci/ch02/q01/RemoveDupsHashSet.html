5   import java.util.HashSet;

7   /**
8    * Implementation of {@link RemoveDups} using a hash-set.
9    *
10   * @param <T> the class of the objects in the list
11   * @author Rosty Kerei
12   */
13  public final class RemoveDupsHashSet<T> implements RemoveDups<T> {
14  
15      @Override
16      public void removeDups(final DoublyLinkedList<T> input) {
17          HashSet<T> uniques = new HashSet<>();
18  
19          DoublyLinkedList<T> node = input;
20  
21          while (node != null) {
22              if (!uniques.add(node.getData())) {
23                  node.getPrevious().setNext(node.getNext());
24              }
25  
26              node = node.getNext();
27          }
28      }
29  }
