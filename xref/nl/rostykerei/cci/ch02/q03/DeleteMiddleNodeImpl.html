3   import nl.rostykerei.cci.datastructure.SinglyLinkedList;
4   
5   /**
6    * Implementation of {@link DeleteMiddleNode}.
7    *
8    * @param <T> the class of the objects in the list
9    * @author Rosty Kerei
10   */
11  public final class DeleteMiddleNodeImpl<T> implements DeleteMiddleNode<T> {
12  
13      @Override
14      public void deleteMiddleNode(final SinglyLinkedList<T> node) {
15          SinglyLinkedList<T> next = node.getNext();
16  
17          if (next == null) {
18              throw new IllegalArgumentException(
19                      "The last node cannot be deleted"
20              );
21          }
22  
23          node.setData(next.getData());
24          node.setNext(next.getNext());
25      }
26  }
