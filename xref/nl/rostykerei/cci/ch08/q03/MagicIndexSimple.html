4    * Simplest implementation of {@link MagicIndex}.

6    * @author Rosty Kerei

8   public final class MagicIndexSimple implements MagicIndex {
9   
10      @Override
11      public Integer findMagicIndex(final int[] input) {
12  
13          for (int i = 0; i < input.length; i++) {
14              if (i == input[i]) {
15                  return i;
16              }
17          }
18  
19          return null;
20      }
21  }
