3   /**
4    * Implementation of {@link MagicIndex} using binary search.
5    *
6    * @author Rosty Kerei
7    */
8   public final class MagicIndexBinarySearch implements MagicIndex {
9   
10      @Override
11      public Integer findMagicIndex(final int[] input) {
12          return binarySearch(input, 0, input.length - 1);
13      }
14  
15      /**
16       * Binary search for magic index. Works on sorted input.
17       *
18       * @param input array to search on
19       * @param start start index
20       * @param end   end index
21       * @return magic index or null if not found
22       */
23      private Integer binarySearch(final int[] input,
24                                   final int start, final int end) {
25          if (start > end) {
26              return null;
27          }
28  
29          int middle = (start + end) >>> 1; // (start + end) / 2
30  
31          if (input[middle] == middle) {
32              return middle;
33          } else if (input[middle] > middle) {
34              return binarySearch(input, start, middle - 1);
35          } else {
36              return binarySearch(input, middle + 1, end);
37          }
38      }
39  }
