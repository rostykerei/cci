3   /**
4    * Brute-force recursive implementataion of {@link TripleStep}.
5    *
6    * @author Rosty Kerei
7    */
8   public class TripleStepImpl implements TripleStep {
9   
10      /**
11       * Possible hops at a time.
12       */
13      private static final int[] HOPS = {1, 2, 3};
14  
15      /**
16       * {@inheritDoc}
17       */
18      @Override
19      public int countPossibleWays(final int stairs) {
20          if (stairs < 0) {
21              return 0;
22          }
23  
24          if (stairs == 0) {
25              return 1;
26          }
27  
28          int result = 0;
29  
30          for (int hop : HOPS) {
31              result += countPossibleWays(stairs - hop);
32          }
33  
34          return result;
35      }
36  }
