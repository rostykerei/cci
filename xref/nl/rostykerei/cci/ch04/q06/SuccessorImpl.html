5   /**
6    * Implementation of {@link Successor}.
7    *
8    * @param <T> the class of the objects in the node
9    * @author Rosty Kerei
10   */
11  public final class SuccessorImpl<T extends Comparable<T>>
12          implements Successor<T> {
13  
14      @Override
15      public BinaryTreeNode<T> findNext(final BinaryTreeNode<T> node) {
16          if (node.getRight() != null) {
17              BinaryTreeNode<T> n = node.getRight();
18  
19              while (n.getLeft() != null) {
20                  n = n.getLeft();
21              }
22  
23              return n;
24          } else {
25              BinaryTreeNode<T> n = node;
26              BinaryTreeNode<T> parent = (BinaryTreeNode<T>) node.getParent();
27  
28              while (parent != null && parent.getLeft() != n) {
29                  n = parent;
30                  parent = (BinaryTreeNode<T>) parent.getParent();
31              }
32  
33              return parent;
34          }
35      }
36  }
