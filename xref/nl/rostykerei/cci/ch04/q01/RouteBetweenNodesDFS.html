3   import nl.rostykerei.cci.datastructure.TreeNode;
4   
5   import java.util.HashSet;
6   import java.util.Set;
7   
8   /**
9    * Implementation of {@link RouteBetweenNodes} using depth-first search.
10   *
11   * @param <T> the class of the objects in the graph
12   * @author Rosty Kerei
13   */
14  public final class RouteBetweenNodesDFS<T> implements RouteBetweenNodes<T> {
15  
16      @Override
17      public boolean isRoutePossible(final TreeNode<T> start,
18                                     final TreeNode<T> finish) {
19          return dfs(start, finish, new HashSet<>());
20      }
21  
22      /**
23       * Recursive depth-first search.
24       *
25       * @param node current node
26       * @param finish finish node
27       * @param seen set of seen nodes
28       * @return true if route can be found, false otherwise
29       */
30      private boolean dfs(final TreeNode<T> node, final TreeNode<T> finish,
31                          final Set<TreeNode> seen) {
32  
33          if (node.equals(finish)) {
34              return true;
35          }
36  
37          seen.add(node);
38  
39          for (TreeNode<T> child : node.getChildren()) {
40              if (!seen.contains(child) && dfs(child, finish, seen)) {
41                  return true;
42              }
43          }
44  
45          return false;
46      }
47  }
