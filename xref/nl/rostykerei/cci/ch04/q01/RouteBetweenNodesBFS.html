7   import java.util.HashSet;

9   /**
10   * Implementation of {@link RouteBetweenNodes} using breadth-first search.
11   *
12   * @param <T> the class of the objects in the graph
13   * @author Rosty Kerei
14   */
15  public final class RouteBetweenNodesBFS<T> implements RouteBetweenNodes<T> {
16  
17      @Override
18      public boolean isRoutePossible(final TreeNode<T> start,
19                                     final TreeNode<T> finish) {
20  
21          HashSet<TreeNode<T>> seen = new HashSet<>();
22          seen.add(start);
23  
24          Queue<TreeNode<T>> queue = new QueueImpl<>();
25          queue.add(start);
26  
27          while (!queue.isEmpty()) {
28              TreeNode<T> currentNode = queue.remove();
29  
30              if (currentNode.equals(finish)) {
31                  return true;
32              }
33  
34              for (TreeNode<T> child : currentNode.getChildren()) {
35                  if (!seen.contains(child)) {
36                      seen.add(child);
37                      queue.add(child);
38                  }
39              }
40          }
41  
42          return false;
43      }
44  }
