4   import nl.rostykerei.cci.datastructure.impl.BinaryTreeNodeImpl;

6   /**
7    * Implementation of {@link MinimalTree}.
8    *
9    * @param <T> the class of the objects in the tree
10   * @author Rosty Kerei
11   */
12  public final class MinimalTreeImpl<T extends Comparable<T>>
13          implements MinimalTree<T> {
14  
15      @Override
16      public BinaryTreeNode<T> buildBST(final T[] input) {
17          return buildNode(input, 0, input.length - 1);
18      }
19  
20      /**
21       * Recursive function to build binary search tree node.
22       *
23       * @param input sorted array of values
24       * @param left  left boundary
25       * @param right right boundary
26       * @return binary search tree node
27       */
28      private BinaryTreeNode<T> buildNode(final T[] input,
29                                          final int left,
30                                          final int right) {
31          if (left < right) {
32              int midPoint = left + (right - left) / 2;
33  
34              return new BinaryTreeNodeImpl<>(
35                      buildNode(input, left, midPoint - 1),
36                      buildNode(input, midPoint + 1, right),
37                      input[midPoint]
38              );
39          } else if (left == right) {
40              return new BinaryTreeNodeImpl<>(input[left]);
41          } else {
42              return null;
43          }
44      }
45  }
