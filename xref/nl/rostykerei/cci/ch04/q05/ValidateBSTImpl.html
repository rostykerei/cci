5   /**
6    * Implementation of {@link ValidateBST}.
7    *
8    * @param <T> the class of the objects in the tree
9    */
10  public final class ValidateBSTImpl<T extends Comparable<T>>
11          implements ValidateBST<T> {
12  
13      @Override
14      public boolean isValidBST(final BinaryTreeNode<T> input) {
15          return validateBSTNode(input, null, null);
16      }
17  
18      /**
19       * Recursively validate BST nodes.
20       *
21       * @param node root node
22       * @param min  allowed minimum or null
23       * @param max  allowed maximum or null
24       * @return true if the node is a valid BST
25       */
26      private boolean validateBSTNode(final BinaryTreeNode<T> node,
27                                      final T min, final T max) {
28          return node == null
29                  || (min == null || node.getData().compareTo(min) >= 0)
30                  && (max == null || node.getData().compareTo(max) <= 0)
31                  && validateBSTNode(node.getLeft(), min, node.getData())
32                  && validateBSTNode(node.getRight(), node.getData(), max);
33      }
34  }
