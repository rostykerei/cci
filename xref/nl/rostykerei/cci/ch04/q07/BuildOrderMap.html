9   import java.util.Map;
10  import java.util.Set;

13  /**
14   * Implementation of {@link BuildOrder} using a standard map.
15   *
16   * @param <T> the class of the project objects
17   * @author Rosty Kerei
18   */
19  public final class BuildOrderMap<T> implements BuildOrder<T> {
20  
21      /**
22       * Map of project dependencies.
23       */
24      private Map<T, Set<T>> dependencyMap = new HashMap<>();
25  
26      @Override
27      public List<T> buildOrder(
28              final Set<T> projects,
29              final Set<Pair<T, T>> dependencies) {
30  
31          for (T project : projects) {
32              dependencyMap.put(project, new HashSet<>());
33          }
34  
35          for (Pair<T, T> pair : dependencies) {
36              dependencyMap.putIfAbsent(pair.getSecond(), new HashSet<>());
37              dependencyMap.get(pair.getSecond()).add(pair.getFirst());
38          }
39  
40          List<T> buildOrder = new ArrayList<>();
41  
42          while (dependencyMap.size() > 0) {
43              boolean found = false;
44  
45              for (T project : projects) {
46                  if (dependencyMap.containsKey(project)
47                          && dependencyMap.get(project).isEmpty()) {
48  
49                      buildOrder.add(project);
50  
51                      dependencyMap.remove(project);
52  
53                      for (Map.Entry<T, Set<T>> entry
54                              : dependencyMap.entrySet()) {
55                          dependencyMap.get(entry.getKey()).remove(project);
56                      }
57  
58                      found = true;
59                  }
60              }
61  
62              if (!found) {
63                  break;
64              }
65          }
66  
67          return buildOrder;
68      }
69  }
