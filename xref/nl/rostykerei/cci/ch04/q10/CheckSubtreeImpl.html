5   /**
6    * Implementation of {@link CheckSubtree} using DFS and
7    * recursive tree comparison.
8    *
9    * @param <T> the class of the objects in the tree
10   * @author Rosty Kerei
11   */
12  public final class CheckSubtreeImpl<T extends Comparable<T>>
13          implements CheckSubtree<T> {
14  
15      @Override
16      public boolean isSubtree(final BinaryTreeNode<T> tree,
17                               final BinaryTreeNode<T> subtree) {
18  
19          return tree != null
20                  && (treeEquals(tree, subtree)
21                  || isSubtree(tree.getRight(), subtree)
22                  || isSubtree(tree.getLeft(), subtree));
23  
24      }
25  
26      /**
27       * Compares two binary trees by data value.
28       *
29       * @param t1 first tree
30       * @param t2 second tree
31       * @return true if trees equal by data value
32       */
33      public boolean treeEquals(final BinaryTreeNode<T> t1,
34                                final BinaryTreeNode<T> t2) {
35          return t1 == t2
36                  || t1 != null
37                  && t2 != null
38                  && t1.getData().equals(t2.getData())
39                  && treeEquals(t1.getLeft(), t2.getLeft())
40                  && treeEquals(t1.getRight(), t2.getRight());
41      }
42  }
