5   import java.util.ArrayList;
6   import java.util.LinkedList;
7   import java.util.List;

9   /**
10   * Implementation of {@link ListOfDepths} using breadth-first search.
11   *
12   * @param <T> the class of the objects in the tree
13   * @author Rosty Kerei
14   */
15  public final class ListOfDepthsBFS<T extends Comparable<T>>
16          implements ListOfDepths<T> {
17  
18      @Override
19      public List<LinkedList<BinaryTreeNode<T>>> getListOfDepth(
20              final BinaryTreeNode<T> input) {
21  
22          List<LinkedList<BinaryTreeNode<T>>> result = new ArrayList<>();
23  
24          LinkedList<BinaryTreeNode<T>> newLevel = new LinkedList<>();
25          newLevel.add(input);
26  
27          while (!newLevel.isEmpty()) {
28              result.add(newLevel);
29  
30              LinkedList<BinaryTreeNode<T>> previousLevel = newLevel;
31              newLevel = new LinkedList<>();
32  
33              for (BinaryTreeNode<T> node : previousLevel) {
34                  if (node.getLeft() != null) {
35                      newLevel.add(node.getLeft());
36                  }
37  
38                  if (node.getRight() != null) {
39                      newLevel.add(node.getRight());
40                  }
41              }
42          }
43  
44          return result;
45      }
46  }
