16      @Test
17      public void testOrder() {
18          Set<String> projects = new HashSet<>();
19          projects.add("A");
20          projects.add("B");
21          projects.add("C");
22          projects.add("D");
23          projects.add("E");
24          projects.add("F");
25  
26          Set<Pair<String, String>> deps = new HashSet<>();
27          deps.add(new PairImpl<>("A", "D"));
28          deps.add(new PairImpl<>("F", "B"));
29          deps.add(new PairImpl<>("B", "D"));
30          deps.add(new PairImpl<>("F", "A"));
31          deps.add(new PairImpl<>("D", "C"));
32  
33          assertTrue(
34                  Arrays.deepEquals(new String[]{"E", "F", "A", "B", "D", "C"},
35                          testInstance.buildOrder(projects, deps).toArray())
36          );
37      }
